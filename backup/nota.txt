<script>
// ── STATE ──────────────────────────────────────────────────────────
let notes = [];
try { notes = JSON.parse(localStorage.getItem('nota_v1') || '[]'); } catch(e){}
let dark = localStorage.getItem('nota_theme') === 'dark';
let panelOpen = false;
let searchOpen = false;
let searchQ = '';
let dragSrc = null;

// ── THEME ──────────────────────────────────────────────────────────
function applyTheme() {
  document.documentElement.setAttribute('data-theme', dark ? 'dark' : 'light');
  document.getElementById('themeBtn').textContent = dark ? '○' : '◑';
}
applyTheme();

document.getElementById('themeBtn').addEventListener('click', () => {
  dark = !dark;
  localStorage.setItem('nota_theme', dark ? 'dark' : 'light');
  applyTheme();
});

// ── PERSIST ────────────────────────────────────────────────────────
function save() { localStorage.setItem('nota_v1', JSON.stringify(notes)); }

// ── PANEL ──────────────────────────────────────────────────────────
function openPanel() {
  panelOpen = true;
  document.getElementById('inputPanel').classList.add('open');
  document.getElementById('backdrop').classList.add('on');
  document.getElementById('fab').classList.add('open');
  document.getElementById('fab').style.zIndex = 150;
  setTimeout(() => document.getElementById('inputArea').focus(), 360);
}
function closePanel() {
  panelOpen = false;
  document.getElementById('inputPanel').classList.remove('open');
  document.getElementById('backdrop').classList.remove('on');
  document.getElementById('fab').classList.remove('open');
  document.getElementById('fab').style.zIndex = 200;
  document.getElementById('inputArea').value = '';
  resetAIBtn();
}

document.getElementById('fab').addEventListener('click', () => {
  panelOpen ? closePanel() : openPanel();
});
document.getElementById('backdrop').addEventListener('click', closePanel);

function resetAIBtn() {
  const btn = document.getElementById('formatBtn');
  btn.disabled = false;
  btn.innerHTML = '<span>⚡</span><span>Format with AI</span>';
}

// ── HELPERS ────────────────────────────────────────────────────────
function esc(s) {
  return (s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}
function timeAgo(ts) {
  if (!ts) return '';
  const d = Date.now() - ts;
  if (d < 60000) return 'just now';
  if (d < 3600000) return Math.floor(d/60000) + 'm ago';
  if (d < 86400000) return Math.floor(d/3600000) + 'h ago';
  return new Date(ts).toLocaleDateString('en',{month:'short',day:'numeric'});
}
const TYPE_ICONS = {
  todo:'✅', note:'📝', task:'📅',
  calculation:'🧮', code:'💻', list:'🛒', raw:'✦'
};

// ── RENDER ─────────────────────────────────────────────────────────
function renderStats() {
  const total = notes.length;
  const done = notes.filter(n => n.completed).length;
  const s = document.getElementById('stats');
  if (!total) { s.innerHTML=''; return; }
  s.innerHTML = `<div class="chip">${total} note${total!==1?'s':''}</div>
    ${done ? `<div class="chip">✓ ${done} done</div>` : ''}`;
}

function buildContent(n) {
  if (n.type === 'todo' && n.todos) {
    return `<div class="note-body"><div class="todo-list">
      ${n.todos.map((t,ti) => `
        <div class="todo-row">
          <div class="tcheck${t.done?' on':''}" data-item="${ti}">${t.done?'✓':''}</div>
          <span style="${t.done?'text-decoration:line-through;opacity:.5':''}">${esc(t.text)}</span>
        </div>`).join('')}
    </div></div>`;
  }
  if (n.type === 'code') {
    return `<div class="code-pre">${esc(n.content)}</div>`;
  }
  return `<div class="note-body">${esc(n.content)}</div>`;
}

function buildCard(n, realIdx) {
  const icon = TYPE_ICONS[n.type] || '✦';
  return `
    <div class="drag-pip"></div>
    <div class="card-top">
      <div class="type-badge">${icon} ${n.type||'note'}</div>
      <div class="card-btns">
        <button class="cbtn btn-complete" data-id="${n.id}" title="${n.completed?'Undo':'Complete'}">${n.completed?'↩':'✓'}</button>
        <button class="cbtn btn-delete" data-id="${n.id}" title="Delete">✕</button>
      </div>
    </div>
    ${buildContent(n)}
    <div class="card-meta">${timeAgo(n.createdAt)}</div>`;
}

function render() {
  renderStats();
  const list = document.getElementById('notesList');
  const shown = searchQ
    ? notes.filter(n => JSON.stringify(n).toLowerCase().includes(searchQ.toLowerCase()))
    : notes;

  if (!shown.length) {
    list.innerHTML = `<div class="empty">
      <span class="empty-orb">✦</span>
      <div class="empty-h">${searchQ ? 'Nothing found' : 'Your canvas is clear'}</div>
      <div class="empty-p">${searchQ ? 'Try a different search' : 'Tap + to capture a thought,\nthen let AI shape it.'}</div>
    </div>`;
    return;
  }

  list.innerHTML = '';
  shown.forEach((n, visIdx) => {
    const realIdx = notes.indexOf(n);
    const wrap = document.createElement('div');
    wrap.className = 'nwrap';
    wrap.innerHTML = `
      <div class="swipe-hint right-bg"><span>✓</span></div>
      <div class="swipe-hint left-bg"><span>🗑</span></div>`;

    const card = document.createElement('div');
    card.className = 'card' + (n.completed ? ' completed' : '');
    card.draggable = true;
    card.dataset.id = n.id;
    card.innerHTML = buildCard(n, realIdx);

    // ── Desktop drag ──
    card.addEventListener('dragstart', e => {
      dragSrc = realIdx;
      card.classList.add('dragging');
      e.dataTransfer.effectAllowed = 'move';
    });
    card.addEventListener('dragend', () => card.classList.remove('dragging'));
    card.addEventListener('dragover', e => { e.preventDefault(); card.classList.add('drag-over'); });
    card.addEventListener('dragleave', () => card.classList.remove('drag-over'));
    card.addEventListener('drop', e => {
      e.preventDefault();
      card.classList.remove('drag-over');
      if (dragSrc !== null && dragSrc !== realIdx) {
        const [moved] = notes.splice(dragSrc, 1);
        notes.splice(realIdx, 0, moved);
        save(); render();
      }
    });

    // ── Touch swipe ──
    let tx0 = 0, ty0 = 0, swiping = false, curX = 0;
    card.addEventListener('touchstart', e => {
      tx0 = e.touches[0].clientX;
      ty0 = e.touches[0].clientY;
      swiping = false; curX = 0;
    }, {passive:true});
    card.addEventListener('touchmove', e => {
      const dx = e.touches[0].clientX - tx0;
      const dy = e.touches[0].clientY - ty0;
      if (!swiping && Math.abs(dx) > Math.abs(dy) && Math.abs(dx) > 8) swiping = true;
      if (swiping) {
        curX = dx;
        card.style.transform = `translateX(${dx}px)`;
        card.style.transition = 'none';
        wrap.querySelector('.right-bg').style.opacity = dx > 0 ? Math.min(1, dx/70) : 0;
        wrap.querySelector('.left-bg').style.opacity = dx < 0 ? Math.min(1, -dx/70) : 0;
      }
    }, {passive:true});
    card.addEventListener('touchend', () => {
      if (!swiping) return;
      card.style.transition = 'transform .25s ease';
      if (curX < -75) {
        card.style.transform = 'translateX(-115%)';
        setTimeout(() => { deleteNote(n.id); }, 240);
      } else if (curX > 75) {
        card.style.transform = 'translateX(0)';
        toggleComplete(n.id);
      } else {
        card.style.transform = 'translateX(0)';
        wrap.querySelector('.right-bg').style.opacity = 0;
        wrap.querySelector('.left-bg').style.opacity = 0;
      }
    });

    // ── Buttons ──
    card.querySelector('.btn-complete').addEventListener('click', () => toggleComplete(n.id));
    card.querySelector('.btn-delete').addEventListener('click', () => deleteNote(n.id));

    // ── Todo checks ──
    card.querySelectorAll('.tcheck').forEach(cb => {
      cb.addEventListener('click', e => {
        const ti = parseInt(e.currentTarget.dataset.item);
        const note = notes.find(x => x.id === n.id);
        if (note?.todos) {
          note.todos[ti].done = !note.todos[ti].done;
          save(); render();
        }
      });
    });

    wrap.appendChild(card);
    list.appendChild(wrap);
  });
}

function toggleComplete(id) {
  const n = notes.find(x => x.id === id);
  if (n) { n.completed = !n.completed; save(); render(); }
}
function deleteNote(id) {
  notes = notes.filter(x => x.id !== id);
  save(); render();
}

// ── SAVE RAW ──────────────────────────────────────────────────────
document.getElementById('saveRawBtn').addEventListener('click', () => {
  const text = document.getElementById('inputArea').value.trim();
  if (!text) return;
  notes.unshift({ id: Date.now(), type: 'raw', content: text, createdAt: Date.now(), completed: false });
  save(); render(); closePanel();
});

// ── AI FORMAT ─────────────────────────────────────────────────────
document.getElementById('formatBtn').addEventListener('click', async () => {
  if (btn.disabled) return;
  const text = document.getElementById('inputArea').value.trim();
  if (text.length > 2000) {
  alert("Input too long (max 2000 characters)");
  return;
}
  if (!text) return;

  const btn = document.getElementById('formatBtn');
  btn.disabled = true;
  btn.innerHTML = '<div class="spinner"></div><span>Thinking…</span>';
  // ✅ ADD THIS BLOCK
const failSafe = setTimeout(() => {
  if (btn.disabled) {
    resetAIBtn();
    alert("AI is taking too long. Try again.");
  }
}, 9000);

  try {
    const suggestions = await callAI(text);
    clearTimeout(failSafe);
    closePanel();
    showSuggestions(suggestions);
  } catch(err) {
    clearTimeout(failSafe);
    alert('Could not reach AI: ' + err.message);
    resetAIBtn();
  }
});

async function callAI(input) {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort(), 8000);

  try {
    const res = await fetch("http://localhost:3000/format", {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({ input }),
      signal: controller.signal
    });

    clearTimeout(timeout);

    if (!res.ok) throw new Error("Server error");

    const data = await res.json();

    if (!data.suggestions || !data.suggestions.length) {
      throw new Error("Empty AI response");
    }

    return data.suggestions;

  } catch (err) {
    clearTimeout(timeout);

    return [{
      type: "note",
      icon: "📝",
      title: "Quick Note",
      content: input
    }];
  }
}

// ── SUGGESTIONS ──────────────────────────────────────────────────
let pendingSuggestions = [];

function showSuggestions(suggestions) {
  pendingSuggestions = suggestions;
  const list = document.getElementById('sugList');
  list.innerHTML = suggestions.map((s, i) => `
    <div class="sug-card" data-idx="${i}">
      <div class="sug-head">
        <span class="sug-icon">${s.icon||'✦'}</span>
        <span class="sug-type">${s.title||s.type}</span>
      </div>
      <div class="sug-preview">${previewSug(s)}</div>
    </div>`).join('');

  document.getElementById('sugOverlay').classList.add('open');

  list.querySelectorAll('.sug-card').forEach(card => {
    card.addEventListener('click', () => {
      const s = pendingSuggestions[parseInt(card.dataset.idx)];
      notes.unshift({
        id: Date.now(), type: s.type,
        content: s.content || '',
        todos: s.todos || null,
        createdAt: Date.now(), completed: false
      });
      save(); render();
      document.getElementById('sugOverlay').classList.remove('open');
    });
  });
}

function previewSug(s) {
  if (s.type === 'todo' && s.todos) {
    return s.todos.slice(0,3).map(t => `☐ ${esc(t.text)}`).join('\n')
      + (s.todos.length > 3 ? `\n+ ${s.todos.length-3} more…` : '');
  }
  if (s.type === 'code') {
    return `<code style="font-family:'SF Mono',monospace;font-size:12px">${esc((s.content||'').slice(0,100))}</code>`;
  }
  return esc((s.content||'').slice(0,130));
}

});
